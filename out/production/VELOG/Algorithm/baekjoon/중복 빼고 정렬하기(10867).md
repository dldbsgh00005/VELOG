# 회고
정렬과 중복 제거가 전부인 문제, 자바 기본 정렬이 최적화가 되어 있고 Stream으로 중복 제거 하여 쉽게 풀림

# 풀이
Java 기본 정렬인 TimSort와 Stream의 distinct를 이용한 손쉬운 풀이.


```
package Algorithm.baekjoon.main;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Main {

    /**
     * Bubble, Selection, Insertion, Quick, Merge 정렬 등이 있다.
     * Java의 기본 정렬은
     * 기본형 : Dual-Pivot Sort
     * 참조형 : TimSort가 있다.
     *
     * 둘 다 성능이 최고급 이므로
     * 자바 기본 정렬을 이용해 풀이하자.
     * **/

    static int N;

    static int[] numbers;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        N = Integer.parseInt(st.nextToken());
        numbers = new int[N];

        st = new StringTokenizer(br.readLine());
        int tokenSize = st.countTokens();
        for(int cir = 0; cir < tokenSize; cir++){
            numbers[cir] = Integer.parseInt(st.nextToken());
        }

        // sort
        sort();
        removeRedundant();

        toStringBuilder();

    }

    public static void removeRedundant(){
            numbers = Arrays.stream(numbers).distinct().toArray();
    }

    public static void sort(){
        Arrays.sort(numbers);
    }

    public static void toStringBuilder(){
        StringBuilder sb = new StringBuilder();

        for(int number : numbers){
            sb.append(number).append(" ");
        }

        System.out.println(sb);
    }
}
```
