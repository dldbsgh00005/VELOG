### 회고
문제 접근을 완전히 잘못해서 삽질한 문제 시작점, 끝점 long으로 받아서 정렬 후 길이 더해주면 되는 문제
정렬 메소드 숙지 필요

### 풀이
1. 2차원 long 배열에 시작점, 끝점을 입력 받는다.
2. Arrays.sort()의 Comparator 기반 시작점, 끝점 기반 정렬 (thenComparing 최고..)
3. 겹치는 부분에 대해 조심해서 총 길이를 더해주면 쉽게 풀리는 문제  

```
package Algorithm.baekjoon.main;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.Comparator;
import java.util.StringTokenizer;

public class Main {

    /**
     * 선 긋기(2170)
     *
     * 길이가 최대 10억, 횟수는 10만 -> BruteForce 절대 불가
     *
     * ! 음수부터 양수까지 랜덤한 숫자들로 이루어진 선(들)
     * 선들의 구분점을 어떻게 구하지
     **/

    static BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
    static StringTokenizer st;

    static int N;
    static long[][] lines;
    static long total = 0;

    public static void main(String[] args) throws IOException {
        initialization();
        sort();
        measureLines();
        printTotal();
    }

    public static void initialization() throws IOException {
        N = Integer.parseInt(br.readLine().trim());
        lines = new long[N][2];

        for(int cir = 0; cir < N; cir++){
            st = new StringTokenizer(br.readLine());
            long cur1 = Long.parseLong(st.nextToken());
            long cur2 = Long.parseLong(st.nextToken());
            if(cur1 > cur2) {lines[cir][0] = cur2; lines[cir][1] = cur1; continue;} // cur1이 cur2보다 크면 0,1번 인덱스 값 체인지
            lines[cir][0] = cur1; lines[cir][1] = cur2; 
        }
    }

    public static void sort(){
        Arrays.sort(lines, Comparator.<long[]>comparingLong( line -> line[0]).thenComparingLong(line -> line[1]));
    }

    public static void measureLines(){
        long st = lines[0][0];
        long end = lines[0][1];
        for(int cir = 0; cir < N; cir++){
            long cur_st = lines[cir][0];
            long cur_end = lines[cir][1];
            if(cur_st <= end) {
                if(end < cur_end) end = cur_end;
            }
            else{
                total += (end - st);
                st = cur_st; end = cur_end;
            }
        }
        total += (end - st);
    }

    public static void printTotal(){
        System.out.println(total);
    }

}
```
